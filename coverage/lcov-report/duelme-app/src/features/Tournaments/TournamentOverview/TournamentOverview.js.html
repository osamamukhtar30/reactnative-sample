import {selectMatchTypeById} from '@duelme/apisdk/dist/slices/games/selectors';
import {MATCH_TYPE_TO_GAME_SETTINGS} from '@duelme/js-constants/dist/games';
import React, {useState} from 'react';
import {useTranslation} from 'react-i18next';
import {LayoutAnimation, View, Platform} from 'react-native';
import {heightPercentageToDP as hp} from 'react-native-responsive-screen';
import {useSelector} from 'react-redux';
import styled from 'styled-components/native';
 
import MatchRules from '../../../components/MatchRules/MatchRules';
import theme from '../../../global-styles/theme';
 
import TournamentDetails from './TournamentDetails';
import TournamentPrize from './TournamentPrize';
import TournamentSchedule from './TournamentSchedule';
 
const AboutTournament = styled.Text`
  font-size: 22px;
  font-family: Quantico-Bold;
  color: ${theme.colors.white};
  letter-spacing: 3px;
  margin-top: 20px;
`;
 
const TournamentDescription = styled.Text`
  font-size: 16px;
  font-family: Quantico-Bold;
  color: ${theme.colors.white};
  margin-top: 10px;
  margin-bottom: 30px;
`;
const Container = styled.View``;
const TabsContainer = styled.View`
  flex-direction: row;
  align-self: center;
`;
const Tab = styled.TouchableOpacity`
  width: 25%;
  height: ${hp(6)}px;
  align-items: center;
  justify-content: center;
  background-color: #050311;
  border-top-color: ${({index, selectedTab}) => (index === selectedTab ? theme.colors.white : theme.colors.grey)};
  border-top-width: 1px;
  border-top-start-radius: 10px;
  border-top-end-radius: 10px;
  border-left-color: ${({index, selectedTab}) => (index === selectedTab ? theme.colors.white : theme.colors.grey)};
  border-left-width: 1px;
  border-right-color: ${({index, selectedTab}) => (index === selectedTab ? theme.colors.white : theme.colors.grey)};
  border-right-width: 1px;
  border-bottom-width: ${({index, selectedTab}) => (index === selectedTab ? 0 : 1)}px;
  border-bottom-color: ${theme.colors.white};
`;
const ContentContainer = styled.View`
  align-self: center;
  width: 100%;
  background-color: #050311;
  border-width: 1px;
  border-color: ${theme.colors.white};
  border-top-width: 0px;
  border-bottom-start-radius: 10px;
  border-bottom-end-radius: 10px;
  min-height: 200px;
  overflow: hidden;
  padding-bottom: 10px;
`;
const TabTitle = styled.Text`
  text-transform: uppercase;
  color: ${({index, selectedTab}) => (index === selectedTab ? theme.colors.white : theme.colors.grey)};
  font-weight: bold;
  font-size: 14px;
  letter-spacing: 2px;
  margin-horizontal: 4px;
`;
 
const Spacer = styled.View`
  height: 20px;
`;
 
const TournamentOverview = ({tournament}) => {
  const [selectedTab, setSelectedTab] = useState(0);
  const [t] = useTranslation('global');
  const matchType = useSelector(state => selectMatchTypeById(state, tournament.matchTypeId));
  const settings = matchType && MATCH_TYPE_TO_GAME_SETTINGS[matchType.name];
  const tabs = [
    {label: t('tournament.details'), value: 0},
    {label: t('tournament.rules'), value: 1},
    {label: t('tournament.prizes'), value: 2},
    {label: t('tournament.schedule'), value: 3},
  ];
  return (
    <View testID="overview">
      <AboutTournament>{t('tournament.about')}</AboutTournament>
      <TournamentDescription testID="description">{tournament?.description}</TournamentDescription>
      <Container>
        <TabsContainer>
          {tabs.map((item, index) => (
            <Tab
              testID="tab"
              index={index}
              selectedTab={selectedTab}
              onPress={() => {
                Eif (Platform.OS === 'ios') {
                  LayoutAnimation.easeInEaseOut();
                }
                setSelectedTab(index);
              }}
              key={item.label}
            >
              <TabTitle adjustsFontSizeToFit numberOfLines={1} index={index} selectedTab={selectedTab}>
                {item.label}
              </TabTitle>
            </Tab>
          ))}
        </TabsContainer>
        <ContentContainer>
          {selectedTab === 0 ? (
            <TournamentDetails settings={settings} tournament={tournament} />
          ) : selectedTab === 1 ? (
            <>
              <Spacer height={2} />
              <MatchRules settings={settings} title={false} noTopMargin={true} designV2={false} />
            </>
          ) : selectedTab === 2 ? (
            <TournamentPrize tournament={tournament} />
          ) : (
            <TournamentSchedule tournament={tournament} />
          )}
        </ContentContainer>
      </Container>
    </View>
  );
};
 
export default TournamentOverview;
