import React, {useState} from 'react';
import {useDispatch} from 'react-redux';
import styled from 'styled-components/native';
import {deleteAccount} from '@duelme/apisdk/dist/slices/account/thunks';
import {useTranslation} from 'react-i18next';
import {Formik} from 'formik';
import * as yup from 'yup';
import {closeModal} from '@duelme/apisdk/dist/slices/modal/native';
 
import Button from '../../../button/Button';
import CustomTextInput from '../../../Inputs/TextInput';
import useSyncEndpointCall from '../../../../utils/syncEndpointCall';
 
import BlurViewContainer from './../../../BlurViewContainer/BlurViewContainer';
import TopSpacer from './../../../Spacer/Top';
import DrawerHeader from './../../../DrawerHeader/DrawerHeader';
import theme from './../../../../global-styles/theme';
 
const ButtonContainer = styled.View`
  width: 90%;
  margin: auto;
  margin-bottom: 20px;
`;
 
const OutContainer = styled.View`
  height: 100%;
`;
 
const WidthContainer = styled.View`
  width: 90%;
  display: flex;
  flex-grow: 1;
  margin: 0px auto;
`;
 
const Title = styled.Text`
  margin: 30px 0px;
  text-align: center;
  color: ${({theme}) => theme.colors.white};
  font-size: 20px;
`;
 
const validationSchema = yup.object().shape({
  password: yup.string().required(),
});
 
const DeleteAccount = () => {
  const [t] = useTranslation('global');
  const syncEndpointCall = useSyncEndpointCall();
  const dispatch = useDispatch();
  const [loading, setLoading] = useState(false);
 
  return (
    <Formik
      initialValues={{
        password: '',
      }}
      onSubmit={() => {}}
      validateOnMount={true}
      validationSchema={validationSchema}
    >
      {({isValid, handleChange, handleBlur, values, errors, touched}) => (
        <BlurViewContainer>
          <OutContainer>
            <DrawerHeader blurContext closeCallback={() => dispatch(closeModal())} />
            <TopSpacer />
            <WidthContainer>
              <Title>{t('delete_account_text')}</Title>
              <CustomTextInput
                testID="passwordConfirmationInput"
                value={values.password}
                label={t('password')}
                secureTextEntry
                onChangeText={handleChange('password')}
                handleBlur={handleBlur('password')}
                isInvalid={!!errors.password}
                feedback={touched.password && errors.password}
              />
              <ButtonContainer>
                <Button
                  testID="confirmationButton"
                  loading={loading}
                  disabled={!isValid || Object.keys(errors).length !== 0}
                  fullWidth
                  solidColor={theme.colors.red}
                  text={t('delete_account')}
                  mode="contained"
                  onPress={async () => {
                    setLoading(true);
                    await syncEndpointCall({
                      avoidOpenLoading: true,
                      avoidCloseLoading: true,
                      loadingText: t('deleting_account'),
                      reduxAction: deleteAccount({password: values.password}),
                      errorText: t('account_was_not_deleted'),
                      successText: t('account_deleted'),
                      errorCallBack: () => setLoading(false),
                      successCallback: () => {
                        setLoading(false);
                      },
                    });
                  }}
                />
              </ButtonContainer>
            </WidthContainer>
          </OutContainer>
        </BlurViewContainer>
      )}
    </Formik>
  );
};
 
export default DeleteAccount;
