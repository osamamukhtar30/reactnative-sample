import React, {useState} from 'react';
import styled from 'styled-components/native';
import {LayoutAnimation} from 'react-native';
import {TouchableOpacity} from 'react-native-gesture-handler';
import {widthPercentageToDP as wp, heightPercentageToDP as hp} from 'react-native-responsive-screen';
import {useTranslation} from 'react-i18next';
import Icon from 'react-native-fontawesome-pro';
 
import TransparentCard from '../TransparentCard/TransparentCard';
import theme from '../../global-styles/theme';
 
const Container = styled.View`
  margin-bottom: 3px;
  ${({designV2}) =>
    designV2 &&
    `
    padding: ${hp(2)}px ${wp(2)}px;
  `}
`;
 
const StyledTouchable = styled(TouchableOpacity)`
  flex-direction: row;
  justify-content: space-between;
  align-items: center;
  ${({designV2}) =>
    !designV2 &&
    `
    background-color: #0e0c11;
    padding: ${wp('3.8%')}px;
    margin-top:10px;
    border-radius: 10px;
  `}
`;
 
const RuleContent = styled.View`
  flex-direction: row;
`;
 
const RuleText = styled.Text`
  font-size: 15px;
  font-weight: bold;
  color: white;
  font-family: Quantico-Bold;
  text-transform: uppercase;
`;
 
const RulesContainer = styled.View`
  padding: ${hp(3)}px ${wp(2)}px;
  ${({designV2}) =>
    designV2 &&
    `
    padding: ${hp(3)}px ${wp(0)}px;
  `}
`;
const RuleLiContainer = styled.View`
  margin: 5px 0px;
  border-width: 1px;
  padding: 10px;
  border-color: gray;
  border-radius: 10px;
  align-items: center;
  flex-direction: row;
`;
const RuleLi = styled.Text`
  color: white;
  font-family: 'Decimal-medium';
  margin-left: 10px;
  margin-right: 15px;
`;
 
const ImageTitile = styled.Text`
  color: white;
  margin: 5px 0px;
  font-weight: bold;
  font-size: 18px;
  ${({designV2}) =>
    designV2 &&
    `
    font-family: 'Quantico-Bold;
    font-weight: 100;
  `}
`;
const BanImage = styled.Image`
  width: ${hp(10)}px;
  height: ${hp(10)}px;
`;
const BanContainer = styled.View`
  align-items: center;
  margin-horizontal: 5px;
`;
const RowContainer = styled.View`
  align-items: center;
  flex-direction: row;
  flex-wrap: wrap;
  justify-content: space-around;
  margin-top: 20px;
`;
const StyledCard = styled(TransparentCard)`
  border-color: ${theme.colors.grey};
  padding: 10px;
  border-radius: 5px;
`;
const Rule = ({title, rules, designV2, iconName, defaultOpen}) => {
  const [opened, setOpened] = useState(defaultOpen ? defaultOpen : false);
  const [t] = useTranslation('global');
 
  return (
    <Container designV2={designV2}>
      <StyledTouchable
        designV2={designV2}
        onPress={() => {
          LayoutAnimation.easeInEaseOut();
          setOpened(!opened);
        }}
      >
        <RuleContent>
          <RuleText designV2={designV2}>{title}</RuleText>
        </RuleContent>
        {!designV2 && <Icon name={'angle-down'} color="white" type="light" size={wp(5)} />}
      </StyledTouchable>
      {opened && (
        <RulesContainer designV2={designV2}>
          {rules.bans ? (
            <>
              <RuleLiContainer>
                <Icon size={20} color="white" name={iconName}></Icon>
                <RuleLi designV2={designV2} key={rules.description}>{` ${t(rules.description)}`}</RuleLi>
              </RuleLiContainer>
              <RowContainer>
                {rules.bans.map(rule => (
                  <BanContainer key={rule.name}>
                    <ImageTitile designV2={designV2}>{`${t(rule.name)}`}</ImageTitile>
                    <StyledCard>
                      <BanImage source={{uri: rule.src}}></BanImage>
                    </StyledCard>
                  </BanContainer>
                ))}
              </RowContainer>
            </>
          ) : (
            rules.map(rule => (
              <RuleLiContainer key={rule}>
                <Icon size={20} color="white" name={iconName}></Icon>
                <RuleLi designV2={designV2}>{`${t(rule)}`}</RuleLi>
              </RuleLiContainer>
            ))
          )}
        </RulesContainer>
      )}
    </Container>
  );
};
 
export default Rule;
