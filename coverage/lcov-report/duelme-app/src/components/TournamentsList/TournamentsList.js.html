import React, {useEffect} from 'react';
import {useSelector, useDispatch} from 'react-redux';
import styled from 'styled-components/native';
import {widthPercentageToDP as wp, heightPercentageToDP as hp} from 'react-native-responsive-screen';
import {useTranslation} from 'react-i18next';
import {selectTournamentsByGameId, selectTournamentsLoading} from '@duelme/apisdk/dist/slices/tournaments/selectors';
import {fetchTournaments} from '@duelme/apisdk/dist/slices/tournaments/thunks';
import {selectSelectedGameId} from '@duelme/apisdk/dist/slices/games/selectors';
import {selectCurrentUser} from '@duelme/apisdk/dist/slices/currentUser/selectors';
import {fetchCurrentUser} from '@duelme/apisdk/dist/slices/account/thunks';
import {TOURNAMENT_STATUS_OPEN_SIGNUP} from '@duelme/js-constants/dist/tournaments';
import Icon from 'react-native-fontawesome-pro';
import {FlatList} from 'react-native';
 
import theme from '../../global-styles/theme';
import LoadingGate from '../../components/LoadingGate/LoadingGate';
 
import Tournament from './Tournament';
import TournamentsEmptyState from './TournamentsEmptyState';
 
const Title = styled.Text`
  text-align: center;
  text-transform: uppercase;
  color: ${({theme}) => theme.colors.white};
  font-size: ${wp(6)}px;
  font-family: 'Quantico-Bold;
`;
 
const Container = styled.View`
  margin-bottom: ${hp(2)}px;
`;
 
const Row = styled.View`
  display: flex;
  justify-content: center;
  align-items: center;
  flex-direction: row;
  ${({useMarginTop}) => useMarginTop && `margin-top: ${hp(3)}px;`}
  margin-bottom: ${hp(1)}px;
`;
 
const IconContainer = styled.TouchableOpacity`
  position: absolute;
  right: ${wp(5)}px;
  padding: 5px;
`;
 
const TournamentsList = ({shouldRefresh, myTournaments, reloadEnabled}) => {
  const dispatch = useDispatch();
  const [t] = useTranslation('global');
  let tournaments, tournamentsLoading, gameId;
  if (myTournaments) {
    const user = useSelector(selectCurrentUser);
    tournaments = user.incomingTournaments;
    tournamentsLoading = !user;
  } else {
    gameId = useSelector(selectSelectedGameId);
    tournaments = useSelector(state => selectTournamentsByGameId(state, gameId));
    tournaments = tournaments.filter(tournament => tournament.status === TOURNAMENT_STATUS_OPEN_SIGNUP);
    tournamentsLoading = useSelector(selectTournamentsLoading);
  }
 
  const fetchData = () => {
    if (!myTournaments) {
      dispatch(fetchTournaments({gameId: gameId}));
    } else {
      dispatch(fetchCurrentUser());
    }
  };
 
  useEffect(() => {
    fetchData();
  }, [gameId, shouldRefresh]);
 
  const renderItem = ({item}) => {
    const tournament = item;
    return <Tournament key={tournament.id} tournament={tournament} />;
  };
 
  return (
    <Container>
      <Row useMarginTop={!myTournaments}>
        <Title>{t(myTournaments ? 'upcoming_tournaments' : 'tournaments')}</Title>
        {reloadEnabled && (
          <IconContainer
            disabled={tournamentsLoading}
            testID={tournamentsLoading ? 'reload-tournaments-loading' : 'reload-tournaments'}
            onPress={fetchData}
          >
            <Icon
              name={'rotate-right'}
              color={tournamentsLoading ? theme.colors.grey : theme.colors.white}
              type="solid"
              size={20}
            />
          </IconContainer>
        )}
      </Row>
      <LoadingGate style={{height: hp(40) + 20}} loading={tournamentsLoading}>
        <>
          <FlatList
            testID="tournament-list"
            initialNumToRender={2}
            data={tournaments}
            renderItem={renderItem}
            horizontal={true}
            showsHorizontalScrollIndicator={false}
          ></FlatList>
          {tournaments?.length === 0 && <TournamentsEmptyState myTournaments={myTournaments} />}
        </>
      </LoadingGate>
    </Container>
  );
};
 
export default TournamentsList;
