import React, {useEffect, useRef} from 'react';
import styled from 'styled-components/native';
import {getFocusedRouteNameFromRoute, useNavigation, useRoute} from '@react-navigation/native';
import {CHILD_STATE} from '@react-navigation/core/src/useRouteCache';
import {heightPercentageToDP as hp} from 'react-native-responsive-screen';
import LinearGradient from 'react-native-linear-gradient';
import {useSelector} from 'react-redux';
import {selectActiveStatus} from '@duelme/apisdk/dist/slices/userStatus/selectors';
import {selectMatchmakingStatus} from '@duelme/apisdk/dist/slices/matchmaking/selectors';
import {STATUS_MM_FINDING_MATCH, STATUS_MM_PENDING_ACCEPT} from '@duelme/js-constants/dist/matchmaking';
import {selectCurrentMatchId} from '@duelme/apisdk/dist/slices/matches/selectors';
 
import theme from '../../global-styles/theme';
import Slider from '../Slider/Slider';
 
import TabButtons from './components/TabButtons';
 
export const tabContainerHeight = hp(10);
export const paddingTop = 0;
export const paddingBottom = 10;
 
const Container = styled.View`
  bottom: 0px;
  padding: 8px;
  padding-bottom: ${hp(2)}px;
  padding-top: ${() => `${paddingTop}px`};
  align-self: center;
  width: 100%;
  position: absolute;
`;
 
const TabBarContainer = styled.View`
  height: ${tabContainerHeight}px;
  border-radius: 50px;
  align-content: center;
  flex-direction: row;
  justify-content: center;
  align-items: center;
  margin: 8px;
`;
 
const BorderGradiantContainer = styled(LinearGradient)`
  width: 100%;
  height: 100%;
  border-radius: 50px;
  position: absolute;
  padding: 1px;
`;
 
const GradiantContainer = styled(LinearGradient)`
  width: 100%;
  height: 100%;
  border-radius: 50px;
`;
 
const TabBar = () => {
  const navigation = useNavigation();
  const route = useRoute();
  const currentScreenName = getFocusedRouteNameFromRoute(route);
  const status = useSelector(selectActiveStatus);
  const statusData = useSelector(state => state.currentUser.status.data);
  const slider = useRef();
  const matchStatus = useSelector(selectMatchmakingStatus);
  const currentMatchId = useSelector(selectCurrentMatchId);
 
  const triggerSlider = async () => {
    slider.current?.openSlider();
  };
 
  useEffect(() => {
    (async () => {
      if (matchStatus !== 'idle') {
        if ([STATUS_MM_PENDING_ACCEPT].includes(matchStatus)) {
          triggerSlider();
        }
      }
    })();
  }, [matchStatus]);
 
  useEffect(() => {
    (async () => {
      if (currentScreenName === 'PlayNow') {
        if (currentMatchId === null && status) {
          triggerSlider();
        } else {
          slider.current?.collapseSlider();
        }
        try {
          const state = route[CHILD_STATE] ?? route.state;
          const params =
            state.routes[
              state.index ?? (typeof state.type === 'string' && state.type !== 'stack' ? 0 : state.routes.length - 1)
            ]?.params;
          if (params?.avoidSlider) {
            slider.current?.collapseSlider();
          }
          // eslint-disable-next-line
        } catch {}
      } else {
        if (currentMatchId === null && matchStatus !== STATUS_MM_FINDING_MATCH) {
          slider.current?.closeSlider();
        }
      }
    })();
  }, [currentScreenName]);
 
  useEffect(() => {
    return () => {
      if (slider && slider.current && slider.current.remove) {
        slider.current.remove();
      }
    };
  }, []);
 
  const buttonsContent = [
    {
      id: 'Home',
      navFunc: async () => {
        await handleNavigation('Home');
      },
      icon: 'house',
      playNow: false,
    },
    {
      id: 'PlayNow',
      navFunc: async () => {
        await handleNavigation('PlayNow');
      },
      icon: 'gamepad-modern',
      playNow: true,
    },
    {
      id: 'FriendsTab',
      navFunc: async () => {
        await handleNavigation('FriendsTab');
      },
      icon: 'users',
      playNow: false,
    },
  ];
 
  const handleNavigation = async screenName => {
    if (screenName === 'Home') {
      if (status === 'matchmaking') {
        slider.current?.collapseSlider();
      } else {
        slider.current?.closeSlider();
      }
      navigation.navigate('Home');
    } else if (screenName === 'PlayNow') {
      if (status === 'tournaments' && currentMatchId === null) {
        navigation.navigate('TournamentsTab', {
          tournamentId: statusData.tournamentId,
        });
      } else if (currentMatchId !== null) {
        navigation.navigate('MatchView', {
          matchId: currentMatchId,
        });
      } else {
        navigation.navigate('PlayNow', {preventFurtherNavigation: false});
      }
    } else if (screenName === 'FriendsTab') {
      if (status === 'matchmaking') {
        slider.current?.collapseSlider();
      } else {
        slider.current?.closeSlider();
      }
      navigation.navigate('FriendsTab');
    }
  };
 
  return (
    <>
      <Slider ref={slider} />
      <Container>
        <TabBarContainer>
          <BorderGradiantContainer
            colors={[theme.colors.vividBlue, theme.colors.trueBlue]}
            start={{x: 0, y: 0}}
            end={{x: 0, y: 1}}
          >
            <GradiantContainer
              colors={[theme.colors.vividBlue, theme.colors.trueBlue, theme.colors.darkPurple]}
              start={{x: 0, y: 9}}
              end={{x: 0, y: 0}}
            ></GradiantContainer>
          </BorderGradiantContainer>
          {buttonsContent.map(data => {
            return (
              <TabButtons
                currentScreenName={currentScreenName}
                status={status}
                currentMatchId={currentMatchId}
                key={data.id}
                id={data.id}
                navFunc={() => data.navFunc()}
                icon={data.icon}
                playNow={data.playNow}
              />
            );
          })}
        </TabBarContainer>
      </Container>
    </>
  );
};
 
export default TabBar;
