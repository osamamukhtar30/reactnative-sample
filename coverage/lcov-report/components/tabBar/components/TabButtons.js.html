import React, {useEffect} from 'react';
import styled from 'styled-components/native';
// import {useRoute} from '@react-navigation/native';
import {Platform, Animated} from 'react-native';
import {widthPercentageToDP as wp, heightPercentageToDP as hp} from 'react-native-responsive-screen';
import Icon from 'react-native-fontawesome-pro';
import {useTranslation} from 'react-i18next';
import {getFocusedRouteNameFromRoute, useRoute} from '@react-navigation/native';
import {BoxShadow} from 'react-native-shadow';
 
import theme from '../../../global-styles/theme';
 
const Container = styled.View`
  position: relative;
  z-index: 1;
  height: 100%;
  width: 33%;
  ${({theme, playNow, shouldAnimate}) =>
    playNow &&
    `
    box-shadow: 0px 0px 6px ${!shouldAnimate ? theme.colors.red : theme.colors.confirmGreen};
    background-color: ${!shouldAnimate ? theme.colors.red : theme.colors.confirmGreen};
    border-color: ${theme.colors.white};
    border-width: 1px;
    border-radius: ${hp(2)}px;
    width: 35%;
    align-items: center;
    height: 110%;
  `}
`;
 
const IconContainer = styled.View`
  box-shadow: ${({theme, selected}) => `0px 0px 5px ${selected ? theme.colors.white : 'transparent'}`};
`;
 
const ShadowContainer = styled.View`
  top: 0px;
  bottom: 0px;
  left: 0px;
  right: 0px;
  position: absolute;
  display: flex;
  justify-content: center;
  align-items: center;
`;
 
const TouchableOpacity = styled.TouchableOpacity`
  width: 100%;
  height: 100%;
  border-radius: ${hp(2)}px;
  display: flex;
  justify-content: center;
  align-items: center;
`;
 
const TabButtons = props => {
  const route = useRoute();
  const currentScreenName = getFocusedRouteNameFromRoute(route);
  // let selected = route.name === props.id;
  const animatedValue = new Animated.Value(0);
  const animatedValueOpacity = new Animated.Value(0);
  const interpolateRotation = animatedValue.interpolate({inputRange: [0, 1], outputRange: ['0deg', '360deg']});
  const interpolateOpacity = animatedValueOpacity.interpolate({inputRange: [0, 1], outputRange: [0, 1]});
  const animatedStyle = {transform: [{rotate: interpolateRotation}]};
  const [t] = useTranslation('global');
  const shouldAnimate = props.currentMatchId || props.status === 'tournaments';
 
  useEffect(() => {
    if (props.id === 'PlayNow' && shouldAnimate && currentScreenName !== 'MatchView') {
      startAnimations();
    } else {
      stopAnimations();
    }
  }, [props, currentScreenName]);
 
  const startAnimations = () => {
    Animated.loop(
      Animated.spring(animatedValue, {toValue: 1, useNativeDriver: true, friction: 20, tension: 1}),
    ).start();
    Animated.loop(
      Animated.sequence([
        Animated.timing(animatedValueOpacity, {toValue: 1, useNativeDriver: true, duration: 1000}),
        Animated.timing(animatedValueOpacity, {toValue: 0, useNativeDriver: true, duration: 1000}),
      ]),
    ).start();
  };
  const stopAnimations = () => {
    animatedValue.stopAnimation();
    animatedValueOpacity.stopAnimation();
  };
 
  const selected = props.id === props.currentScreenName;
 
  return (
    <>
      {props.playNow && Platform.OS === 'android' && (
        <BoxShadow
          setting={{
            width: wp(31),
            height: hp(10.5),
            color: !shouldAnimate ? theme.colors.red : theme.colors.confirmGreen,
            radius: hp(2),
            border: 10,
            opacity: 1,
            style: {
              position: 'absolute',
              zIndex: 0,
              marginLeft: 10,
              marginRight: 10,
              marginTop: 20,
              marginBottom: 20,
            },
          }}
        ></BoxShadow>
      )}
      <Container playNow={props.playNow} shouldAnimate={shouldAnimate}>
        <TouchableOpacity onPress={() => props.navFunc()}>
          <Animated.View style={animatedStyle}>
            <IconContainer selected={selected}>
              <Icon name={props.icon} color={theme.colors.white} type={'solid'} size={props.playNow ? 50 : 24} />
              {Platform.OS === 'android' && selected && (
                <ShadowContainer>
                  <BoxShadow
                    setting={{
                      width: (props.playNow ? 50 : 24) / 1.5,
                      height: (props.playNow ? 50 : 24) / 1.5,
                      color: theme.colors.white,
                      border: 15,
                      radius: (props.playNow ? 25 : 12) / 1.5,
                      opacity: 0.2,
                    }}
                  />
                </ShadowContainer>
              )}
            </IconContainer>
          </Animated.View>
          {(props.currentMatchId || props.status === 'tournaments') && props.id === 'PlayNow' && (
            <Animated.Text
              style={{
                opacity: interpolateOpacity,
                color: theme.colors.white,
                fontFamily: 'Quantico-Bold',
                textAlign: 'center',
                textTransform: 'capitalize',
                lineHeight: 15,
              }}
            >
              {props.status === 'tournaments'
                ? props.currentMatchId
                  ? t('user_status_text_tournament_matches')
                  : t('user_status_text_tournaments')
                : props.currentMatchId
                ? t('user_status_text_matches')
                : ''}
            </Animated.Text>
          )}
        </TouchableOpacity>
      </Container>
    </>
  );
};
 
export default TabButtons;
